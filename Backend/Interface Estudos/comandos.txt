Ativar flask:
  python -m venv venv
  .\venv\Scripts\Activate.ps1
  pip install flask
  flask run

Acesso ao MySQL:
  pip install sqlalchemy
  pip install mysql-connector-python
  mysql -u root -p

Migrar banco:
  $env:FLASK_APP = "app.py"
  pip install Flask-Migrate
  flask db init
  flask db migrate -m "Adicionar tabela Post"
  flask db upgrade
